arr = [2,4,6,1,0,-3]
target = 2

arr.sort()
l = len(arr)
left = 0
right = l-1
mid = (left + right) // 2
ans = []
flag = 0
while left <= right:
    if arr[mid] == target:
        ans.append(True)
        ans.append(mid)
        flag = 1
        break
    elif arr[mid] < target:
        left = mid + 1
        mid = (left + right) // 2
    else:
        right = mid - 1
        mid = (left + right) // 2
if flag == 0:
    ans.append(False)
    ans.append(None)
print(ans)
